\paragraph{KUBC homogenization.}
Let $\varepsilon_c:=10^{-3}$ be a characteristic strain scale. For a prescribed macroscopic (symmetric) strain tensor $\mathbf{E}$ we impose the affine boundary field $\mathbf{u}_D(x)=\mathbf{E}x$ and solve the kinematic uniform boundary condition by symmetric Nitsche coupling on $\partial\Omega$ (with $\theta=1$): find $\mathbf{u}\in[H^1(\Omega)]^d$ such that for all $\mathbf{v}\in[H^1(\Omega)]^d$
\begin{equation}\label{eq:kubc-nitsche-weak}
\begin{aligned}
(\boldsymbol{\sigma}(\mathbf{u},\rho,\mathbf{A}),\,\boldsymbol{\varepsilon}(\mathbf{v}))_{\Omega}
&\;-\;\langle \boldsymbol{\sigma}(\mathbf{u},\rho,\mathbf{A})\,\mathbf{n},\,\mathbf{v}\rangle_{\partial\Omega}
\;-\;\langle \boldsymbol{\sigma}(\mathbf{v},\rho,\mathbf{A})\,\mathbf{n},\,\mathbf{u}\rangle_{\partial\Omega}
\;+\;\Big\langle \frac{\gamma}{h}\,\mathbf{u},\,\mathbf{v}\Big\rangle_{\partial\Omega}
\\
&=\;-\;\langle \boldsymbol{\sigma}(\mathbf{v},\rho,\mathbf{A})\,\mathbf{n},\,\mathbf{u}_D\rangle_{\partial\Omega}
\;+\;\Big\langle \frac{\gamma}{h}\,\mathbf{u}_D,\,\mathbf{v}\Big\rangle_{\partial\Omega},
\end{aligned}
\end{equation}
where $\boldsymbol{\varepsilon}(\cdot)$ and $\boldsymbol{\sigma}(\cdot,\rho,\mathbf{A})$ are given by \eqref{eq:mech-eps-nd}–\eqref{eq:stress-law-nd}, $h$ is a local facet size, and the penalty is chosen as $\gamma=\alpha\,E_{\mathrm{ref}}$ with $\alpha>0$ and a reference modulus $E_{\mathrm{ref}}$. The Hill–Mandel work equivalence holds:
\begin{equation}\label{eq:hill-mandel-nd}
\overline{\boldsymbol{\sigma}}:\mathbf{E}
\;=\;\frac{1}{|\Omega|}\int_{\Omega}\boldsymbol{\sigma}(\mathbf{u},\rho,\mathbf{A}):\boldsymbol{\varepsilon}(\mathbf{u})\,\mathrm{d}x,
\qquad
\overline{\boldsymbol{\sigma}}:=\frac{1}{|\Omega|}\int_{\Omega}\boldsymbol{\sigma}(\mathbf{u},\rho,\mathbf{A})\,\mathrm{d}x.
\end{equation}

\paragraph{Six canonical tests and the apparent tensor.}
We drive six strain cases of engineering magnitude $\varepsilon_{\mathrm{eng}}>0$:
\[
\mathbf{E}^{(1)}=\varepsilon_{\mathrm{eng}}\mathbf{e}_1\!\otimes\!\mathbf{e}_1,\quad
\mathbf{E}^{(2)}=\varepsilon_{\mathrm{eng}}\mathbf{e}_2\!\otimes\!\mathbf{e}_2,\quad
\mathbf{E}^{(3)}=\varepsilon_{\mathrm{eng}}\mathbf{e}_3\!\otimes\!\mathbf{e}_3,
\]
\[
\mathbf{E}^{(4)}=\tfrac12\varepsilon_{\mathrm{eng}}(\mathbf{e}_2\!\otimes\!\mathbf{e}_3+\mathbf{e}_3\!\otimes\!\mathbf{e}_2),\quad
\mathbf{E}^{(5)}=\tfrac12\varepsilon_{\mathrm{eng}}(\mathbf{e}_1\!\otimes\!\mathbf{e}_3+\mathbf{e}_3\!\otimes\!\mathbf{e}_1),\quad
\mathbf{E}^{(6)}=\tfrac12\varepsilon_{\mathrm{eng}}(\mathbf{e}_1\!\otimes\!\mathbf{e}_2+\mathbf{e}_2\!\otimes\!\mathbf{e}_1),
\]
so that shear uses the engineering convention $\gamma_{ij}=2E_{ij}$. Let $s^{(k)}\in\mathbb{R}^6$ be the Voigt vector of $\overline{\boldsymbol{\sigma}}^{(k)}$ in the order $(11,22,33,23,13,12)$. The apparent stiffness is assembled columnwise as
\begin{equation}\label{eq:kubc-C-nd}
\mathbf{C}_{\mathrm{app}}(:,k)\;=\;\frac{s^{(k)}}{\varepsilon_{\mathrm{eng}}},\qquad k=1,\dots,6,
\end{equation}
followed by minor–major symmetrization. Rotating $\mathbf{C}_{\mathrm{app}}$ to a fabric-aligned basis via $C'_{pqrs}=R_{pi}R_{qj}R_{rk}R_{sl}C_{ijkl}$ and inverting $\mathbf{S}'=(\mathbf{C}')^{-1}$ yields principal moduli $E_i$, shear $G_{ij}$, and Poisson ratios $\nu_{ij}$ (Voigt convention).

\paragraph{Penalty scaling.}
In \eqref{eq:kubc-nitsche-weak} the penalty enters as $(\gamma/h)\langle\cdot,\cdot\rangle_{\partial\Omega}$ and must be comparable to a representative shear/Young modulus to guarantee coercivity. We set $\gamma=\alpha\,E_{\mathrm{ref}}$. A robust choice is $E_{\mathrm{ref}}\approx \sup_{\Omega}E(\rho)$ (or a safe upper percentile) so that a mesh-independent $\alpha=\mathcal{O}(10^2)$ suffices; using $E_{\mathrm{ref}}\equiv E_0$ is consistent provided $\alpha$ is chosen large enough to dominate the largest $E(\rho)$ realized by the current $(\rho,\mathbf{A})$ field. This keeps the KUBC bilinear form uniformly coercive.
