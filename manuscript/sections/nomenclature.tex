\label{sec:nomenclature}
\begin{longtable}{@{} l p{0.75\textwidth} @{}}
\toprule
\textbf{Symbol} & \textbf{Description} \\
\midrule

\toprule
\textbf{Symbol} & \textbf{Description} \\
\midrule
\endhead
\bottomrule
\endfoot
%% --- Primary state variables ---
$\mathbf{u}$ & Displacement field [mm] \\
$\rho$ & Apparent bone density [g/cm$^3$] \\
$S$ & Remodeling stimulus (dimensionless) \\
$\mathbf{A}$ & Fabric tensor (symmetric, positive-definite, $\det=1$) \\
$\mathbf{L}$ & Log-fabric tensor, $\mathbf{L}=\ln\mathbf{A}$, trace-free \\
%% --- Mechanical quantities ---
$\boldsymbol{\varepsilon}$ & Linearized strain tensor \\
$\boldsymbol{\sigma}$ & Cauchy stress tensor [MPa] \\
$\mathbb{C}$ & Fourth-order elasticity tensor [MPa] \\
$\psi$ & Strain energy density (SED) [MPa] \\
$\psi_{\mathrm{day}}$ & Daily SED stimulus (cycle-weighted) [MPa] \\
$\bar{\mathbf{Q}}$ & Cycle-averaged deviatoric-stress outer-product \\
%% --- Constitutive / fabric ---
$E_0$, $\nu_0$ & Reference Young's modulus [MPa] and Poisson's ratio \\
$p$, $q$ & Stiffness exponents (trabecular / cortical) \\
$k_0$, $l_0$, $\mu_0$ & Zysset--Curnier fabric-elasticity parameters \\
$\lambda_i$ & Eigenvalue of $\mathbf{A}$ ($i=1,2,3$) \\
$\mathbf{P}_i$ & Eigenprojector of $\mathbf{A}$ (Sylvester construction) \\
%% --- Stimulus submodel ---
$\psi_{\mathrm{ref}}$ & Reference stimulus (mechanostat set-point) \\
$\psi_{\mathrm{ref}}^{\mathrm{trab}}$, $\psi_{\mathrm{ref}}^{\mathrm{cort}}$ & Trabecular and cortical reference stimuli \\
$r$ & Cortical-to-trabecular reference ratio, $r=\psi_{\mathrm{ref}}^{\mathrm{cort}}/\psi_{\mathrm{ref}}^{\mathrm{trab}}$ \\
$\tau_S$ & Stimulus time constant [day] \\
$D_S$ & Stimulus diffusivity [mm$^2$/day] \\
$\kappa_S$ & Stimulus saturation gain \\
$S_{\max}$ & Stimulus saturation level \\
%% --- Density submodel ---
$k_{\mathrm{form}}$, $k_{\mathrm{res}}$ & Formation and resorption rate constants [g/cm$^3$/day] \\
$\rho_{\min}$, $\rho_{\max}$ & Density bounds [g/cm$^3$] \\
$\alpha_{\mathrm{surf}}(\rho)$ & Surface-availability factor \\
$D_\rho$ & Density diffusivity [mm$^2$/day] \\
%% --- Fabric submodel ---
$\tau_A$ & Fabric relaxation time [day] \\
$D_A$ & Fabric diffusivity [mm$^2$/day] \\
$a(\bar{\mathbf{Q}})$ & Fabric activity function \\
%% --- Solver and time integration ---
$\Delta t$ & Time step size [day] \\
$\beta$ & Picard damping factor \\
$m$ & Anderson history depth \\
$\lambda$ & Anderson Tikhonov regularization parameter \\
$q_{\mathrm{contr}}$ & Contraction ratio (fixed-point) \\
$q_{\mathrm{contr,on}}$, $q_{\mathrm{contr,off}}$ & Anderson on/off switching thresholds \\
$r_{\mathrm{Pic}}$ & Scaled relative Picard residual \\
%% --- Geometry and loading ---
$\Omega$ & Computational domain \\
$\Gamma_D$, $\Gamma_N$ & Dirichlet and Neumann boundary subsets \\
$\mathbf{n}$ & Outward unit normal \\
$\alpha_{\mathrm{front}}$, $\alpha_{\mathrm{sag}}$ & Hip force frontal-plane and sagittal-plane angles [deg] \\
$\sigma_{\deg}$ & Gaussian traction patch angular width [deg] \\
%% --- Diffusion lengths ---
$\ell_S$, $\ell_\rho$, $\ell_A$ & Characteristic diffusion lengths [mm] \\
$\alpha$ & Mesh-based diffusion length factor, $\ell=\alpha h$ \\
$h$ & Characteristic element size [mm] \\
%% --- Smooth operators ---
$\operatorname{smax}$, $\operatorname{smin}$, $\operatorname{sclamp}$ & Smooth max, min, clamp operators \\
$\varepsilon$ & Smoothing width \\
\end{longtable}
