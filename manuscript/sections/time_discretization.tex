\section*{Time discretization}
\label{sec:be-time}

Let $0=t^0<t^1<\dots<t^N=T$ be a (possibly non-uniform) partition of the time interval, and define the step sizes
\[
\Delta t^n := t^{n}-t^{n-1}>0,\qquad n=1,\dots,N.
\]
For any time-dependent field $X$ we write $X^n \approx X(\cdot,t^n)$.
At each time level we seek the coupled remodeling state $\mathbf{x}^n:=(\mathbf{L}^n,S^n,\rho^n)$.
Mechanics is quasi-static (no time derivative); the evolution equations for $S$, $\rho$, and $\mathbf{L}$ are advanced by Backward--Euler steps of size $\Delta t^n$.

\paragraph{Mechanics (quasi-static).}
Given $(\rho^n,\mathbf{L}^n)$ we solve the equilibrium problem \eqref{eq:mech-strong-nd} to obtain the displacement field $\mathbf{u}^n$ and the mechanical drivers $\psi_{\mathrm{day}}^n$ and $\bar{\mathbf{Q}}^n$ defined in \eqref{eq:stim-psi-nd} and \eqref{eq:M-def-nd}.

\paragraph{Stimulus (Backward--Euler).}
Discretizing \eqref{eq:stim-strong-nd} by Backward--Euler gives the linear update: find $S^n$ such that
\begin{equation}\label{eq:be-stim}
\tau_S\frac{S^{n}-S^{n-1}}{\Delta t^n}\;-\;\tau_S D_S\,\Delta S^{n}\;+\;S^{n}
\;=\; S_{\max}\,\tanh\!\Big(\delta_{\mathrm{eff}}(\psi_{\mathrm{day}}^{n},\rho^{n})/\kappa_S\Big)\quad\text{in }\Omega,
\end{equation}
with boundary and initial conditions from \eqref{eq:stim-bc-ic-nd}. For $D_S=0$ the Laplacian term is absent.

\paragraph{Density (Backward--Euler).}
Let $S_+^n:=S_+(S^n)$ and $S_-^n:=S_-(S^n)$, where $S_\pm$ are defined in \eqref{eq:smooth-splits-nd}. Discretizing \eqref{eq:density-strong-nd} by Backward--Euler and evaluating the surface-availability factor $\alpha_{\mathrm{surf}}$ from \eqref{eq:Asurf-nd} explicitly as $\alpha_{\mathrm{surf}}(\rho)\approx \alpha_{\mathrm{surf}}(\rho^{n-1})$ yields
\begin{equation}\label{eq:be-rho}
\frac{\rho^{n}-\rho^{n-1}}{\Delta t^n}\;-\;\nabla\!\cdot\!\big(\mathbf{D}_\rho\nabla\rho^{n}\big)
=\;\alpha_{\mathrm{surf}}(\rho^{n-1})\Big(
k_{\mathrm{form}}\,S_+^{n}\Big(1-\frac{\rho^{n}}{\rho_{\max}}\Big)
+k_{\mathrm{res}}\,S_-^{n}\Big(1-\frac{\rho^{n}}{\rho_{\min}}\Big)
\Big)\quad\text{in }\Omega,
\end{equation}
with boundary and initial conditions from \eqref{eq:density-bc-ic-nd}.

\paragraph{Direction/fabric (Backward--Euler).}
Discretizing \eqref{eq:L-strong-nd} by Backward--Euler yields the linear relaxation--diffusion update: find $\mathbf{L}^n$ such that
\begin{equation}\label{eq:be-A}
c_A\frac{\mathbf{L}^{n}-\mathbf{L}^{n-1}}{\Delta t^n}\;-\;c_A D_A\,\Delta \mathbf{L}^{n}\;+\;\frac{c_A\,a(\bar{\mathbf{Q}}^{n})}{\tau_A}\,\mathbf{L}^{n}
\;=\;\frac{c_A\,a(\bar{\mathbf{Q}}^{n})}{\tau_A}\,\mathbf{L}_M(\bar{\mathbf{Q}}^{n})\quad\text{in }\Omega,
\end{equation}
with boundary and initial conditions from \eqref{eq:L-bc-ic-nd}. The Laplacian acts componentwise on $\mathbf{L}$, and for $D_A=0$ the diffusion term is absent.

\paragraph{Coupled solve at time level $t^n$.}
The submodels are coupled through the mechanics drivers $(\psi_{\mathrm{day}}^n,\bar{\mathbf{Q}}^n)$ and through the density dependence in the stimulus. We obtain $\mathbf{x}^n$ by the block Gauss--Seidel fixed-point scheme described in \S\ref{sec:fp-aa}.

\paragraph{Adaptive time stepping and error control.}
When adaptive control is enabled, each \emph{attempted} step of size $\Delta t^n$ uses an explicit Adams--Bashforth predictor
as an initial guess for the coupled implicit Backward--Euler solve on the state vector $\mathbf x:=(\mathbf{L},S,\rho)$.
The predictor and controller operate on the global DOF vector obtained by concatenating the DOFs of the integrated fields.

\emph{Discrete rates and AB predictor.}
After an accepted step, the stored discrete rate for each field is updated by the backward difference quotient
\[
\dot{\mathbf x}^{n}\approx \frac{\mathbf x^{n}-\mathbf x^{n-1}}{\Delta t^n},
\]
and the previous rate is stored for multi-step prediction. For an attempted step from $t^{n-1}$ to $t^{n}=t^{n-1}+\Delta t^n$, set
$r_n:=\Delta t^n/\Delta t^{n-1}$. If at least two previous accepted steps are available, the variable-step AB2 predictor is
\begin{equation}\label{eq:ab2-predict}
\mathbf{x}_{\mathrm{pred}}^{n}
=\mathbf{x}^{n-1}+\Delta t^n\left(\left(1+\tfrac12 r_n\right)\dot{\mathbf x}^{n-1}-\tfrac12 r_n\,\dot{\mathbf x}^{n-2}\right).
\end{equation}
Otherwise, we fall back to AB1,
$\mathbf x_{\mathrm{pred}}^{n}=\mathbf x^{n-1}+\Delta t^n\dot{\mathbf x}^{n-1}$; on the very first step the history is zero,
so the predictor is constant $\mathbf x_{\mathrm{pred}}^{1}=\mathbf x^{0}$.

\emph{WRMS predictor--corrector error.}
After the coupled Backward--Euler solve yields the corrected state $\mathbf{x}^{n}$ (by the fixed-point scheme in \S\ref{sec:fp-aa}),
the time-stepper computes the weighted RMS error norm
\begin{equation}\label{eq:wrms-error}
e^{n}
=\left(\frac{1}{N_{\mathrm{tot}}}\sum_{i=1}^{N_{\mathrm{tot}}}
\left(\frac{x_i^{n}-x_{\mathrm{pred},i}^{n}}{\mathrm{rtol}_t\,\max(|x_i^{n}|,|x_{\mathrm{pred},i}^{n}|)+\mathrm{atol}_t}\right)^{\!2}\right)^{\!1/2},
\end{equation}
where $N_{\mathrm{tot}}$ is the total number of DOFs across the integrated fields.
For numerical robustness we use $e_{\mathrm{safe}}:=\max(e_{\min},e^{n})$ with a small $e_{\min}>0$ when the error enters a division or a power.

\emph{Acceptance, rejection, and timestep update.}
Let $\eta\in(0,1)$ be a safety factor, $g_{\min}\in(0,1)$ and $g_{\max}>1$ the shrink/growth caps, and let $k_{\mathrm{exp}},k_P,k_I>0$ be controller gains.
Also let $\Delta t_{\min},\Delta t_{\max}$ be hard bounds and define the clamp
\[
\operatorname{clamp}_{[\Delta t_{\min},\Delta t_{\max}]}(h):=\min(\Delta t_{\max},\max(\Delta t_{\min},h)).
\]

\begin{itemize}
\item \textbf{Solver failure.} If the coupled nonlinear solve does not converge, the step is rejected and
\[
\Delta t_{\mathrm{next}} := \operatorname{clamp}_{[\Delta t_{\min},\Delta t_{\max}]}\!\big(\Delta t^n\,g_{\min}\big),
\]
and the PI memory is reset to $e_{\mathrm{prev}}:=1$.

\item \textbf{First accepted step.} The first step is accepted and $e_{\mathrm{prev}}\leftarrow e^{1}$.
If $e^{1}>1$, the next step size is reduced \emph{without redoing the first solve}:
\[
\Delta t_{\mathrm{next}} := \operatorname{clamp}_{[\Delta t_{\min},\Delta t_{\max}]}\!\left(\Delta t^1\,
\max\!\big(g_{\min},\min(0.9,\eta(1/e^{1})^{1/k_{\mathrm{exp}}})\big)\right).
\]
If $e^{1}\le 1$, then $\Delta t_{\mathrm{next}}:=\operatorname{clamp}_{[\Delta t_{\min},\Delta t_{\max}]}(\Delta t^1)$.

\item \textbf{Time-error rejection.} For subsequent steps, if $e^{n}>1$ the step is rejected and
\[
\Delta t_{\mathrm{next}} := \operatorname{clamp}_{[\Delta t_{\min},\Delta t_{\max}]}\!\left(\Delta t^n\,
\max\!\big(g_{\min},\min(0.9,\eta(1/e^{n})^{1/k_{\mathrm{exp}}})\big)\right),
\]
and $e_{\mathrm{prev}}$ is \emph{not} updated (it always stores the last accepted-step error).

\item \textbf{Acceptance and Gustafsson PI update.} If $e^{n}\le 1$, the step is accepted and the next step size is computed by
\[
\Delta t_{\mathrm{next}} := \operatorname{clamp}_{[\Delta t_{\min},\Delta t_{\max}]}\!\left(\Delta t^n\,
\min\!\Big(g_{\max},\max\!\big(g_{\min},\,\eta\,(e^{n})^{-k_I}(e_{\mathrm{prev}}/e^{n})^{k_P}\big)\Big)\right),
\]
with the I-only variant on the second accepted step:
\[
\Delta t_{\mathrm{next}} := \operatorname{clamp}_{[\Delta t_{\min},\Delta t_{\max}]}\!\left(\Delta t^n\,
\min\!\Big(g_{\max},\max\!\big(g_{\min},\,\eta\,(e^{n})^{-k_I}\big)\Big)\right).
\]
If $e^{n}<0.5$ we additionally prevent accidental shrink by enforcing the multiplicative factor to be at least $1$.
Finally, on acceptance the PI memory is updated as $e_{\mathrm{prev}}\leftarrow e^{n}$.
\end{itemize}
